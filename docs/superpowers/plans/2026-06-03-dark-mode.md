# Dark Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an app-wide dark mode with a sun/moon header toggle that follows the OS preference on first load and persists the user's explicit choice thereafter.

**Architecture:** The app already uses Tailwind's `darkMode: ['class']` plus shadcn-style CSS-variable tokens, so defining a `.dark {}` variable block flips the whole UI. A `zustand` + `persist` theme store toggles the `.dark` class on `<html>`; an inline script in `index.html` applies the saved theme before first paint to avoid a flash.

**Tech Stack:** React 18, Vite, TypeScript, Tailwind 3.4, zustand (+persist middleware), lucide-react, Vitest + Testing Library.

**Working directory:** All paths are relative to `frontend/`. Run all commands from `frontend/`. The feature branch `feature/dark-mode` already exists and is checked out.

- [ ] **Step 4: Write the store**

Create `src/stores/theme.store.ts`:

```ts
import { create } from 'zustand'
import { persist, createJSONStorage } from 'zustand/middleware'

export type Theme = 'light' | 'dark' | 'system'

const MEDIA = '(prefers-color-scheme: dark)'

/** Resolve a stored theme to an effective 'light' | 'dark'. */
export function resolveTheme(theme: Theme): 'light' | 'dark' {
  if (theme !== 'system') return theme
  if (typeof window === 'undefined' || !window.matchMedia) return 'light'
  return window.matchMedia(MEDIA).matches ? 'dark' : 'light'
}

/** Toggle the `.dark` class on <html> to match the resolved theme. */
export function applyTheme(theme: Theme): void {
  if (typeof document === 'undefined') return
  document.documentElement.classList.toggle('dark', resolveTheme(theme) === 'dark')
}

interface ThemeState {
  theme: Theme
  setTheme: (theme: Theme) => void
  toggleTheme: () => void
}

export const useThemeStore = create<ThemeState>()(
  persist(
    (set, get) => ({
      theme: 'system',
      setTheme: (theme) => {
        applyTheme(theme)
        set({ theme })
      },
      toggleTheme: () => {
        const next: Theme = resolveTheme(get().theme) === 'dark' ? 'light' : 'dark'
        applyTheme(next)
        set({ theme: next })
      },
    }),
    {
      name: 'inventory-theme',
      storage: createJSONStorage(() => localStorage),
      partialize: (state) => ({ theme: state.theme }),
      onRehydrateStorage: () => (state) => {
        applyTheme(state?.theme ?? 'system')
      },
    },
  ),
)

// Keep "system" users in sync with live OS theme changes.
if (typeof window !== 'undefined' && window.matchMedia) {
  window.matchMedia(MEDIA).addEventListener('change', () => {
    if (useThemeStore.getState().theme === 'system') applyTheme('system')
  })
}
```

## Task 3: Flash-prevention script

**Files:**
- Modify: `index.html`
- Modify: `src/main.tsx`

- [ ] **Step 1: Add the inline script to `index.html`**

In `index.html`, inside `<head>`, immediately after the `<title>…</title>` line, add:

```html
    <script>
      // Apply the saved theme before paint to avoid a flash of light mode.
      // Reads the zustand-persist payload: {"state":{"theme":"dark"},...}
      (function () {
        try {
          var raw = localStorage.getItem('inventory-theme')
          var theme = raw ? JSON.parse(raw).state.theme : 'system'
          var dark =
            theme === 'dark' ||
            (theme === 'system' &&
              window.matchMedia('(prefers-color-scheme: dark)').matches)
          if (dark) document.documentElement.classList.add('dark')
        } catch (e) {}
      })()
    </script>
```

- [ ] **Step 2: Side-effect import the store at startup**

In `src/main.tsx`, add this import alongside the existing imports (after `import './index.css'`):

This guarantees the store rehydrates and registers its OS listener at app startup, independent of when the Header mounts.

Replace the contents of `src/components/layout/Header.tsx` with:

```tsx
import { Menu, Moon, Sun } from 'lucide-react'
import { Button } from '@/components/ui/button'
import { useAuthStore } from '@/stores/auth.store'
import { useThemeStore, resolveTheme } from '@/stores/theme.store'

interface HeaderProps {
  title?: string
  onMenuClick?: () => void
}

export function Header({ title = 'Dashboard', onMenuClick }: HeaderProps) {
  const { user } = useAuthStore()
  const theme = useThemeStore((s) => s.theme)
  const toggleTheme = useThemeStore((s) => s.toggleTheme)
  const isDark = resolveTheme(theme) === 'dark'
  const label = isDark ? 'Switch to light mode' : 'Switch to dark mode'

  return (
    <header className="flex h-16 items-center border-b bg-background px-6">
      <Button
        variant="ghost"
        size="icon"
        className="mr-4"
        onClick={onMenuClick}
      >
        <Menu className="h-5 w-5" />
        <span className="sr-only">Menu</span>
      </Button>
      <h2 className="flex-1 text-xl font-semibold">{title}</h2>
      <Button
        variant="ghost"
        size="icon"
        onClick={toggleTheme}
        aria-label={label}
        title={label}
      >
        {isDark ? <Sun className="h-5 w-5" /> : <Moon className="h-5 w-5" />}
        <span className="sr-only">{label}</span>
      </Button>
      {user && (
        <span className="ml-4 text-sm text-muted-foreground">
          Olá, {user.name}
        </span>
      )}
    </header>
  )
}
```
